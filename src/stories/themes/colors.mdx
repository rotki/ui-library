import { baseColors, baseColorsIntensities, contextColors, contextColorsIntensities } from '@/consts/colors';

# Colors
#### You can use these classes with prefix `bg-rui-`, `text-rui-`, `border-rui-`.

&nbsp;
### Base Colors

<div className="max-w-full overflow-y-scroll">
  <table>
    <tbody>
      <tr>
        <th>Color</th>
        {baseColorsIntensities.map(intensity => (
          <th key={intensity}>{intensity}</th>
        ))}
      </tr>
      {baseColors.map(color => (
        <tr key={color}>
          <td>
            {color}
          </td>
          {baseColorsIntensities.map(intensity => (
            <td key={`${color}-${intensity}`} className="sb-unstyled text-center">
              <div className={`w-12 h-12 mx-auto rounded shadow bg-rui-${color}-${intensity}`}></div>
              <div className="font-bold mt-2"><code>{`${color}-${intensity}`}</code></div>
            </td>
          ))}
        </tr>
      ))}
    </tbody>
  </table>
</div>
